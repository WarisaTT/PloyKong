SSL Certificate — Wildcard HTTPS

SSL/TLS
⚠️
Wildcard certificate (*.ploykong.com) ต้องใช้ DNS-01 challenge เท่านั้น ไม่สามารถใช้ HTTP-01 ได้
1

ติดตั้ง Certbot + DNS Plugin

bash 
apt install -y certbot python3-certbot-dns-digitalocean

# ถ้าใช้ Cloudflare:
pip3 install certbot-dns-cloudflare
2

ขอ Wildcard SSL Certificate

bash 
# วิธี Manual DNS (ใช้ได้กับทุก registrar)
certbot certonly \
  --manual \
  --preferred-challenges dns \
  -d ploykong.com \
  -d "*.ploykong.com" \
  --email [email protected] \
  --agree-tos

# Certbot จะให้สร้าง TXT record ชื่อ:
# _acme-challenge.ploykong.com → [token]
# ไปสร้างใน DNS panel แล้วกด Enter

# Cert จะอยู่ที่:
/etc/letsencrypt/live/ploykong.com/fullchain.pem
/etc/letsencrypt/live/ploykong.com/privkey.pem
3

Database — MySQL Setup & Backup

MYSQL 8
1

ตรวจสอบ Schema

-- ดู tables ทั้งหมด
SHOW TABLES;
-- users, portfolios, sections, analytics, refresh_tokens, ai_chat_logs, files

-- ดู seed data
SELECT email, plan FROM users LIMIT 5;
SELECT slug, title FROM portfolios LIMIT 5;
2

ตั้งค่า Auto Backup (Cron)

bash 
mkdir -p /opt/backups/mysql

# สร้าง backup script
cat > /opt/backup-mysql.sh << 'EOF'
#!/bin/bash
DATE=$(date +%Y%m%d_%H%M%S)
FILE="/opt/backups/mysql/ploykong_${DATE}.sql.gz"
docker exec ploykong-mysql mysqldump \
  -u root -p${MYSQL_ROOT_PASSWORD} ploykong \
  | gzip > "$FILE"
# ลบ backup เก่ากว่า 7 วัน
find /opt/backups/mysql -name "*.sql.gz" -mtime +7 -delete
echo "Backup: $FILE"
EOF
chmod +x /opt/backup-mysql.sh

# เพิ่มใน crontab (backup ทุกคืน 2AM)
echo "0 2 * * * /opt/backup-mysql.sh >> /var/log/mysql-backup.log 2>&1" | crontab -

Routing Architecture

routing

Pipeline Flow

flow 
# เมื่อ push ไป main branch:
1. ✅  go test ./...          # Unit tests + race detector
2. 🏗️  npm run build           # Vue.js production build
3. 📤  upload artifact         # Store dist/ เป็น GitHub artifact
4. 🚀  SSH to VPS             # rsync + docker compose deploy
5. 🐳  docker compose up -d   # Rolling restart
6. 🔍  health check           # curl /health → verify 200
7. 🧹  docker image prune     # Clean old images

# เวลาเฉลี่ย: ~4-6 นาที
4
